### 获取一个变量的类型, 整体类型/子类型

```ts
const res = [

    {

      "avatar": "http://dummyimage.com/100x100",

      "group": "elit nulla Excepteur",

      "is_active": false,

      "nickname": "liang1",

      "gender": 28

    },

    {

      "avatar": "http://dummyimage.com/100x100",

      "group": "elit nulla Excepteur",

      "is_active": false,

      "nickname": "liang2",

      "gender": 21

    },

    {

      "avatar": "http://dummyimage.com/100x100",

      "group": "elit nulla Excepteur",

      "is_active": false,

      "nickname": "liang3",

      "gender": 20

    }
    
]

/** 获取整个res的类型 */
type DataType = typeof res
//type DataType = {  
//	avatar: string;  
//	group: string;  
//	is_active: boolean;  
//	nickname: string;  
//	gender: number;  
//}[]

/** 获取res类型下的每一个对象的类型 */
type ItemType = typeof res[number]
//type ItemType = {  
//	avatar: string;  
//	group: string;  
//	is_active: boolean;  
//	nickname: string;  
//	gender: number;  
//}

/** 获取数组对象下对象的所有属性类型 */
type ItemAttrType = keyof typeof res[number] 
// type IType = "avatar" | "group" | "is_active" | "nickname" | "gender"
```

### 获取一个函数的类型--ReturnType

```ts
function IFunction() {

    return {

      a: 'a',

      b: 'b',

      c: 'c'

    }

}

type IFun = ReturnType<typeof IFunction>

//type IFun = {  
//a: string;  
//b: string;  
//c: string;  
//}
```
